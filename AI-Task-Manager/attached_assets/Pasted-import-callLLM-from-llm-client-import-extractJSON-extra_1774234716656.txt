import { callLLM } from "../llm/client";
import { extractJSON, extractTextContent } from "../llm/parser";
import { FileTool } from "./fileTool";
import { SearchTool } from "./searchTool";

export interface Optimization {
  file: string;
  type: "performance" | "bundle" | "memory" | "query" | "render" | "network" | "security";
  description: string;
  impact: "high" | "medium" | "low";
  before: string;
  after: string;
  applied: boolean;
}

export class CodeOptimizeTool {
  private fileTool: FileTool;
  private searchTool: SearchTool;

  constructor(fileTool: FileTool, searchTool: SearchTool) {
    this.fileTool = fileTool;
    this.searchTool = searchTool;
  }

  async analyzeAndOptimize(options?: {
    targets?: string[];
    autoApply?: boolean;
    types?: Optimization["type"][];
  }): Promise<Optimization[]> {
    const optimizations: Optimization[] = [];
    const files = options?.targets || (await this.searchTool.searchFiles("*.{ts,tsx}")).slice(0, 20);

    for (const file of files) {
      const content = this.fileTool.readFile(file);
      if (!content.success || !content.output) continue;

      // Skip small files
      if (content.output.length < 200) continue;

      const response = await callLLM({
        system: `You are a performance optimization expert. Analyze code and find optimizations.
Focus on:
- React: memo, useMemo, useCallback, lazy loading, code splitting
- Queries: N+1, missing indexes, unnecessary fetches
- Bundle: tree-shaking, dynamic imports, dead code
- Memory: leaks, unnecessary closures, large objects
- Network: batching, caching, compression

Respond with JSON:
{
  "optimizations": [
    {
      "type": "performance|bundle|memory|query|render|network|security",
      "description": "what to optimize",
      "impact": "high|medium|low",
      "before": "the code to replace (exact match)",
      "after": "optimized replacement code"
    }
  ]
}
Only include optimizations where you're confident the change is correct.`,
        messages: [{ role: "user", content: `Optimize ${file}:\n\`\`\`\n${content.output.slice(0, 4000)}\n\`\`\`` }],
      });

      const parsed = extractJSON(extractTextContent(response.content));
      if (!parsed?.optimizations) continue;

      for (const opt of parsed.optimizations) {
        const optimization: Optimization = {
          file,
          ...opt,
          applied: false,
        };

        if (options?.autoApply && content.output.includes(opt.before)) {
          const newContent = content.output.replace(opt.before, opt.after);
          this.fileTool.createFile(file, newContent);
          optimization.applied = true;
        }

        optimizations.push(optimization);
      }
    }

    return optimizations;
  }

  // Quick optimizations that are always safe
  async quickOptimize(): Promise<{ filesOptimized: number; changes: string[] }> {
    const changes: string[] = [];
    let count = 0;

    // 1. Add React.memo to components without it
    const components = await this.searchTool.searchContent("export function ", {
      filePattern: "**/*.tsx",
      maxResults: 30,
    });

    // 2. Find console.logs in production code
    const consoleLogs = await this.searchTool.findAndReplace(
      "console.log(",
      "// console.log(",
      { filePattern: "src/**/*.{ts,tsx}", dryRun: false }
    );
    if (consoleLogs.length > 0) {
      count += consoleLogs.length;
      changes.push(`Commented out ${consoleLogs.reduce((s, r) => s + r.replacements, 0)} console.log calls`);
    }

    // 3. Add loading="lazy" to images
    const images = await this.searchTool.searchContent('<img ', {
      filePattern: "**/*.{tsx,jsx,html}",
      maxResults: 20,
    });

    for (const img of images) {
      if (!img.text.includes('loading=')) {
        const content = this.fileTool.readFile(img.file);
        if (content.success) {
          const updated = (content.output || "").replace(
            /<img\s/g,
            '<img loading="lazy" '
          );
          if (updated !== content.output) {
            this.fileTool.createFile(img.file, updated);
            count++;
            changes.push(`Added lazy loading to images in ${img.file}`);
          }
        }
      }
    }

    return { filesOptimized: count, changes };
  }
}
