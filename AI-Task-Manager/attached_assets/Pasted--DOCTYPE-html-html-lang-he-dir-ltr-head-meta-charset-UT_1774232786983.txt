    function initTerminal() {
      term = new window.Terminal({
        theme: {
          background: '#000000', foreground: '#c9d1d9', cursor: '#7c3aed',
          selectionBackground: '#264f78',
          black: '#484f58', red: '#f85149', green: '#3fb950', yellow: '#d29922',

        fontSize: 13,
        cursorBlink: true,
        allowProposedApi: true,
      });

      const fitAddon = new window.FitAddon.FitAddon();
      term.loadAddon(fitAddon);
      term.open(document.getElementById('terminalContainer'));
      fitAddon.fit();

      term.writeln('\x1b[1;35m╔═══════════════════════════════╗\x1b[0m');
      term.writeln('\x1b[1;35m║    🤖 Kobi Agent Terminal     ║\x1b[0m');
      term.writeln('\x1b[1;35m╚═══════════════════════════════╝\x1b[0m');
      term.writeln('');

      // Input handling
      let currentLine = '';
      term.onKey(({ key, domEvent }) => {
        if (domEvent.key === 'Enter') {
          term.writeln('');
          if (currentLine.trim()) executeCommand(currentLine.trim());
          currentLine = '';
          term.write('\x1b[32m❯\x1b[0m ');
        } else if (domEvent.key === 'Backspace') {
          if (currentLine.length > 0) {
            currentLine = currentLine.slice(0, -1);
            term.write('\b \b');
          }
        } else if (key.length === 1) {
          currentLine += key;
          term.write(key);
        }
      });

      term.write('\x1b[32m❯\x1b[0m ');
      window.addEventListener('resize', () => fitAddon.fit());
    }

    async function executeCommand(cmd) {
      try {
        const res = await fetch('/api/terminal', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ command: cmd }),
        });
        const data = await res.json();
        if (data.stdout) term.writeln(data.stdout);
        if (data.stderr) term.writeln('\x1b[31m' + data.stderr + '\x1b[0m');
      } catch (err) {
        term.writeln('\x1b[31mError: ' + err.message + '\x1b[0m');
      }
      term.write('\x1b[32m❯\x1b[0m ');
    }

    // ═══════════ FILE TREE ═══════════
    async function refreshFiles() {
      try {
        const res = await fetch('/api/files?recursive=true');
        const data = await res.json();
        renderFileTree(data.output || '');
      } catch { document.getElementById('fileTree').innerHTML = '<div style="padding:12px;color:var(--dim)">Error loading files</div>'; }
    }

    function renderFileTree(output) {
      const container = document.getElementById('fileTree');
      if (!output) { container.innerHTML = '<div style="padding:12px;color:var(--dim)">Empty workspace</div>'; return; }

      const lines = output.split('\n').filter(Boolean);
      container.innerHTML = lines.map(line => {
        const isDir = line.endsWith('/');
        const depth = (line.match(/\//g) || []).length;
        const name = line.split('/').pop().replace('/', '');
        const icon = isDir ? '📁' : getFileIcon(name);
        const indent = '<span class="file-indent"></span>'.repeat(Math.max(0, depth - 1));

        return `<div class="file-item ${isDir ? 'dir' : ''}" onclick="${isDir ? '' : `openFile('${line}')`}" title="${line}">
          ${indent}<span class="file-icon">${icon}</span><span>${name}</span>
        </div>`;
      }).join('');
    }

    function getFileIcon(name) {
      const ext = name.split('.').pop();
      const icons = { ts:'🔷', tsx:'⚛', js:'🟡', jsx:'⚛', json:'📋', html:'🌐', css:'🎨', md:'📝', py:'🐍', sql:'🗃', yml:'⚙', yaml:'⚙', env:'🔒', sh:'⌨', prisma:'💎' };
      return icons[ext] || '📄';
    }

    // ═══════════ FILE OPERATIONS ═══════════
    async function openFile(filePath) {
      try {
        const res = await fetch(`/api/files/read?path=${encodeURIComponent(filePath)}`);
        const data = await res.json();
        if (!data.success) return;

        const ext = filePath.split('.').pop();
        const langMap = { ts:'typescript', tsx:'typescript', js:'javascript', jsx:'javascript', json:'json', html:'html', css:'css', md:'markdown', py:'python', sql:'sql', yml:'yaml', yaml:'yaml', sh:'shell', prisma:'prisma' };
        const language = langMap[ext] || 'plaintext';

        // Create or get model
        let model;
        if (openFiles.has(filePath)) {
          model = openFiles.get(filePath).model;
          model.setValue(data.output);
        } else {
          model = monaco.editor.createModel(data.output, language);
          openFiles.set(filePath, { model, content: data.output, language });
        }

        editor.setModel(model);
        activeFile = filePath;

        updateTabs();
        document.getElementById('sbFile').textContent = filePath;
        document.getElementById('sbLang').textContent = language;

        // Highlight in tree
        document.querySelectorAll('.file-item').forEach(el => el.classList.remove('active'));
        document.querySelectorAll('.file-item').forEach(el => {
          if (el.title === filePath) el.classList.add('active');
        });
      } catch {}
    }

    async function saveCurrentFile() {
      if (!activeFile || !editor) return;
      const content = editor.getValue();
      try {
        await fetch('/api/files/write', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ path: activeFile, content }),
        });
        term?.writeln(`\x1b[32m✓ Saved ${activeFile}\x1b[0m`);
      } catch {}
    }

    function updateTabs() {
      const container = document.getElementById('editorTabs');
      container.innerHTML = Array.from(openFiles.entries()).map(([path]) => {
        const name = path.split('/').pop();
        const isActive = path === activeFile;
        return `<div class="tab ${isActive ? 'active' : ''}" onclick="openFile('${path}')">
          <span>${name}</span>
          <span class="tab-close" onclick="event.stopPropagation();closeTab('${path}')">✕</span>
        </div>`;
      }).join('');
    }

    function closeTab(path) {
      const file = openFiles.get(path);
      if (file) file.model.dispose();
      openFiles.delete(path);
      if (activeFile === path) {
        const remaining = Array.from(openFiles.keys());
        if (remaining.length > 0) openFile(remaining[remaining.length - 1]);
        else { editor.setModel(monaco.editor.createModel('', 'plaintext')); activeFile = null; }
      }
      updateTabs();
    }

    // ═══════════ CHAT / AGENT ═══════════
    async function sendMessage() {
      const input = document.getElementById('chatInput');
      const text = input.value.trim();
      if (!text) return;

      addChatMessage('user', text);
      input.value = '';
      input.style.height = 'auto';
      document.getElementById('sbAgent').textContent = 'Agent: Working...';

      try {
        const res = await fetch('/api/task', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ task: text }),
        });
        const data = await res.json();
        addChatMessage('agent', data.success ? `✅ ${data.summary}` : `⚠️ ${data.summary}`);
        refreshFiles();
      } catch (err) {
        addChatMessage('agent', `❌ ${err.message}`);
      }

      document.getElementById('sbAgent').textContent = 'Agent: Idle';
    }

    function addChatMessage(type, text) {
      const el = document.createElement('div');
      el.className = `msg ${type}`;
      // Format code blocks
      el.innerHTML = text.replace(/`([^`]+)`/g, '<code style="background:rgba(0,0,0,.3);padding:1px 4px;border-radius:3px;font-family:var(--mono);font-size:12px">$1</code>');
      const container = document.getElementById('chatMessages');
      container.appendChild(el);
      container.scrollTop = container.scrollHeight;
    }

    // ═══════════ WEBSOCKET ═══════════
    function connectWS() {
      const proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
      ws = new WebSocket(`${proto}//${location.host}/ws`);

      ws.onopen = () => {
        document.getElementById('wsDot').className = 'status-dot on';
        document.getElementById('wsStatus').textContent = 'Connected';
      };

      ws.onclose = () => {
        document.getElementById('wsDot').className = 'status-dot off';
        document.getElementById('wsStatus').textContent = 'Disconnected';
        setTimeout(connectWS, 3000);
      };

      ws.onmessage = e => {
        const event = JSON.parse(e.data);
        if (event.type === 'log' && term) {
          term.writeln(event.data.message);
        }
        if (event.type === 'stream' && term) {
          term.writeln(event.data);
        }
        if (event.type === 'step_complete' || event.type === 'step_failed') {
          addChatMessage('system', `${event.type === 'step_complete' ? '✅' : '❌'} Step ${event.data.stepId}: ${event.data.result?.type || ''}`);
        }
        if (event.type === 'complete') {
          addChatMessage('agent', event.data.summary);
          refreshFiles();
          document.getElementById('sbAgent').textContent = 'Agent: Idle';
        }
        if (event.type === 'phase') {
          document.getElementById('sbAgent').textContent = `Agent: ${event.data.phase}`;
        }
      };
    }
