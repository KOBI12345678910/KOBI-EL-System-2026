-- USERS
CREATE TABLE IF NOT EXISTS users (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
username VARCHAR(50) UNIQUE NOT NULL,
email VARCHAR(200) UNIQUE NOT NULL,
password_hash TEXT NOT NULL,
first_name VARCHAR(100),
last_name VARCHAR(100),
role VARCHAR(50) DEFAULT 'viewer' CHECK (role IN ('admin','manager','sales','production','warehouse','finance','hr','installer','viewer')),
employee_id UUID REFERENCES employees(id),
permissions JSONB DEFAULT '{}',
last_login TIMESTAMP,
is_active BOOLEAN DEFAULT true,
created_at TIMESTAMP DEFAULT NOW(),
updated_at TIMESTAMP DEFAULT NOW()
);-- CUSTOMERS
CREATE TABLE IF NOT EXISTS customers (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
customer_number VARCHAR(20) UNIQUE,
customer_type VARCHAR(20) DEFAULT 'company' CHECK (customer_type IN ('individual','company','government','non_profit')),
company_name VARCHAR(200),
trade_name VARCHAR(200),
tax_id VARCHAR(20),
vat_number VARCHAR(20),
first_name VARCHAR(100),
last_name VARCHAR(100),
email VARCHAR(200),
phone VARCHAR(20),
mobile VARCHAR(20),
fax VARCHAR(20),
website VARCHAR(300),
billing_address_street VARCHAR(300),
billing_address_city VARCHAR(100),
billing_address_zip VARCHAR(20),
billing_address_country VARCHAR(100) DEFAULT 'ישראל',
shipping_address_street VARCHAR(300),
shipping_address_city VARCHAR(100),
shipping_address_zip VARCHAR(20),
shipping_address_country VARCHAR(100) DEFAULT 'ישראל',
shipping_same_as_billing BOOLEAN DEFAULT true,
industry VARCHAR(100),
customer_group VARCHAR(20) DEFAULT 'regular' CHECK (customer_group IN ('vip','regular','wholesale','retail','government')),
payment_terms_id UUID,
credit_limit INTEGER DEFAULT 0,
current_balance INTEGER DEFAULT 0,
currency VARCHAR(3) DEFAULT 'ILS',
price_list_id UUID,
discount_percent DECIMAL(5,2) DEFAULT 0,
sales_rep_id UUID,
territory VARCHAR(100),
source VARCHAR(30) DEFAULT 'other',
lifecycle_stage VARCHAR(20) DEFAULT 'active',
tags JSONB DEFAULT '[]',
custom_fields JSONB DEFAULT '{}',
notes TEXT,
is_active BOOLEAN DEFAULT true,
created_by UUID,
created_at TIMESTAMP DEFAULT NOW(),
updated_at TIMESTAMP DEFAULT NOW()
);-- CONTACTS

created_at TIMESTAMP DEFAULT NOW(),
updated_at TIMESTAMP DEFAULT NOW()
);-- SUPPLIERS
CREATE TABLE IF NOT EXISTS suppliers (

updated_at TIMESTAMP DEFAULT NOW()
);-- RAW MATERIALS
CREATE TABLE IF NOT EXISTS raw_materials (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
material_number VARCHAR(30) UNIQUE,
name VARCHAR(200) NOT NULL,
name_en VARCHAR(200),
description TEXT,
category VARCHAR(30) NOT NULL CHECK (category IN ('iron','steel','aluminum','stainless_steel','glass','wood','paint','welding_materials','hardware','sealants','accessories','packaging','other')),
sub_category VARCHAR(100),
material_type VARCHAR(100),
specification TEXT,
grade VARCHAR(50),

unit_of_measure VARCHAR(10) DEFAULT 'unit' CHECK (unit_of_measure IN ('kg','m','m2','unit','liter','roll','sheet','bar','tube','set')),

supplier_id UUID REFERENCES suppliers(id),
alternate_suppliers JSONB DEFAULT '[]',
last_purchase_price INTEGER DEFAULT 0,
average_cost INTEGER DEFAULT 0,
standard_cost INTEGER DEFAULT 0,
currency VARCHAR(3) DEFAULT 'ILS',
min_stock_level DECIMAL(12,3) DEFAULT 0,
max_stock_level DECIMAL(12,3) DEFAULT 0,
reorder_point DECIMAL(12,3) DEFAULT 0,
reorder_quantity DECIMAL(12,3) DEFAULT 0,
current_stock DECIMAL(12,3) DEFAULT 0,
warehouse_location VARCHAR(50),

customs_code VARCHAR(20),
country_of_origin VARCHAR(100),
image_url VARCHAR(500),
barcode VARCHAR(50),
technical_specs JSONB DEFAULT '{}',
tags JSONB DEFAULT '[]',
notes TEXT,
is_active BOOLEAN DEFAULT true,
created_by UUID,
created_at TIMESTAMP DEFAULT NOW(),
updated_at TIMESTAMP DEFAULT NOW()
);-- PRODUCTS
CREATE TABLE IF NOT EXISTS products (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
product_number VARCHAR(30) UNIQUE,
name VARCHAR(200) NOT NULL,
name_en VARCHAR(200),
description TEXT,
category VARCHAR(30) NOT NULL CHECK (category IN ('railing','gate','fence','pergola','door','staircase','window','curtain_wall','custom','accessory')),

unit_of_measure VARCHAR(10) DEFAULT 'unit',
standard_price INTEGER DEFAULT 0,
cost_price INTEGER DEFAULT 0,

image_urls JSONB DEFAULT '[]',
technical_drawing_url VARCHAR(500),
certifications JSONB DEFAULT '[]',
tags JSONB DEFAULT '[]',
is_active BOOLEAN DEFAULT true,
created_by UUID,
created_at TIMESTAMP DEFAULT NOW(),
updated_at TIMESTAMP DEFAULT NOW()
);-- SALES ORDERS

updated_at TIMESTAMP DEFAULT NOW()
