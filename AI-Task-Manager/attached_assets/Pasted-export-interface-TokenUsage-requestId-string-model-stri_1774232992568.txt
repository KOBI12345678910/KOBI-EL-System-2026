export interface TokenUsage {
  requestId: string;
  model: string;
  inputTokens: number;
  outputTokens: number;
  totalTokens: number;
  cost: number;
  timestamp: Date;
  taskId?: string;
  phase?: string;
}

export class TokenTracker {
  private usage: TokenUsage[] = [];
  private totalCost = 0;
  private totalTokens = 0;

  // Pricing per million tokens
  private pricing: Record<string, { input: number; output: number }> = {
    "claude-sonnet-4-20250514": { input: 3, output: 15 },
    "claude-opus-4-20250514": { input: 15, output: 75 },
    "claude-haiku-4-5-20251001": { input: 0.8, output: 4 },
  };

  record(params: {
    model: string;
    inputTokens: number;
    outputTokens: number;
    taskId?: string;
    phase?: string;
  }): TokenUsage {
    const pricing = this.pricing[params.model] || { input: 3, output: 15 };
    const cost = (params.inputTokens * pricing.input + params.outputTokens * pricing.output) / 1_000_000;

    const usage: TokenUsage = {
      requestId: `req_${Date.now()}`,
      model: params.model,
      inputTokens: params.inputTokens,
      outputTokens: params.outputTokens,
      totalTokens: params.inputTokens + params.outputTokens,
      cost,
      timestamp: new Date(),
      taskId: params.taskId,
      phase: params.phase,
    };

    this.usage.push(usage);
    this.totalCost += cost;
    this.totalTokens += usage.totalTokens;

    return usage;
  }

  getStats(): {
    totalRequests: number;
    totalTokens: number;
    totalCost: number;
    avgTokensPerRequest: number;
    avgCostPerRequest: number;
    byModel: Record<string, { requests: number; tokens: number; cost: number }>;
    byTask: Record<string, { requests: number; tokens: number; cost: number }>;
    last24h: { tokens: number; cost: number; requests: number };
  } {
    const byModel: Record<string, { requests: number; tokens: number; cost: number }> = {};
    const byTask: Record<string, { requests: number; tokens: number; cost: number }> = {};
    const dayAgo = Date.now() - 86400000;
    let last24h = { tokens: 0, cost: 0, requests: 0 };

    for (const u of this.usage) {
      // By model
      if (!byModel[u.model]) byModel[u.model] = { requests: 0, tokens: 0, cost: 0 };
      byModel[u.model].requests++;
      byModel[u.model].tokens += u.totalTokens;
      byModel[u.model].cost += u.cost;

      // By task
      if (u.taskId) {
        if (!byTask[u.taskId]) byTask[u.taskId] = { requests: 0, tokens: 0, cost: 0 };
        byTask[u.taskId].requests++;
        byTask[u.taskId].tokens += u.totalTokens;
        byTask[u.taskId].cost += u.cost;
      }

      // Last 24h
      if (u.timestamp.getTime() > dayAgo) {
        last24h.tokens += u.totalTokens;
        last24h.cost += u.cost;
        last24h.requests++;
      }
    }

    return {
      totalRequests: this.usage.length,
      totalTokens: this.totalTokens,
      totalCost: Math.round(this.totalCost * 10000) / 10000,
      avgTokensPerRequest: this.usage.length > 0 ? Math.round(this.totalTokens / this.usage.length) : 0,
      avgCostPerRequest: this.usage.length > 0 ? Math.round((this.totalCost / this.usage.length) * 10000) / 10000 : 0,
      byModel,
      byTask,
      last24h: {
        ...last24h,
        cost: Math.round(last24h.cost * 10000) / 10000,
      },
    };
  }

  estimateCost(task: string): { estimatedTokens: number; estimatedCost: number } {
    // Estimate based on task complexity
    const words = task.split(/\s+/).length;
    const isComplex = task.includes("full") || task.includes("complete") || task.includes("entire");
    const multiplier = isComplex ? 8 : 4;
    const estimatedTokens = words * 100 * multiplier;
    const pricing = this.pricing["claude-sonnet-4-20250514"];
    const estimatedCost = (estimatedTokens * 0.6 * pricing.input + estimatedTokens * 0.4 * pricing.output) / 1_000_000;

    return {
      estimatedTokens,
      estimatedCost: Math.round(estimatedCost * 10000) / 10000,
    };
  }

  getRecent(limit = 50): TokenUsage[] {
    return this.usage.slice(-limit);
  }

  reset(): void {
    this.usage = [];
    this.totalCost = 0;
    this.totalTokens = 0;
  }
}
