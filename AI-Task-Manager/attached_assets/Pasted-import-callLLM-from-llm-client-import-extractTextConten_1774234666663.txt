export class AgentModeManager {
  private currentMode: AgentMode = "power";
  private features: { webSearch: boolean; extendedThinking: boolean; appTesting: boolean; codeOptimizations: boolean } = {
    webSearch: true,
    extendedThinking: false,
    appTesting: true,
    codeOptimizations: true,
  };

  setMode(mode: AgentMode): ModeConfig {
    this.currentMode = mode;
    return this.getConfig();
  }

  getMode(): AgentMode { return this.currentMode; }

  getConfig(): ModeConfig {
    const base = { ...AGENT_MODES[this.currentMode] };
    base.webSearch = this.features.webSearch;
    base.extendedThinking = this.features.extendedThinking;
    base.autoTest = this.features.appTesting;
    base.autoOptimize = this.features.codeOptimizations;
    return base;
  }

  toggleFeature(feature: keyof typeof this.features, value?: boolean): boolean {
    this.features[feature] = value ?? !this.features[feature];
    return this.features[feature];
  }

  getFeatures() { return { ...this.features }; }

  estimateCost(taskComplexity: "simple" | "medium" | "complex"): { tokens: number; cost: number; duration: string } {
    const config = this.getConfig();
    const baseTokens = { simple: 5000, medium: 25000, complex: 80000 }[taskComplexity];
    const tokens = baseTokens * config.costMultiplier;
    const costPer1M = config.model.includes("haiku") ? 4.8 : 18;
    const cost = (tokens * costPer1M) / 1_000_000;
    const minutes = { simple: 1, medium: 5, complex: 15 }[taskComplexity] * (config.costMultiplier > 3 ? 0.4 : 1);
    return { tokens: Math.round(tokens), cost: Math.round(cost * 1000) / 1000, duration: `~${Math.round(minutes)} min` };
  }
}
