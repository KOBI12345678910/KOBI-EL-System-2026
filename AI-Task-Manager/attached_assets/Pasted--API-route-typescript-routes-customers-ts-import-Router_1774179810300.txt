// routes/customers.ts
import { Router, Request, Response } from 'express';
import { db } from '../db';

const router = Router();

// GET /api/customers - List with pagination, search, filter
router.get('/', async (req: Request, res: Response) => {
  try {
    const {
      page = '1',
      limit = '20',
      search = '',
      sort = 'created_at',
      order = 'desc',
      status,
      customer_group,
      customer_type
    } = req.query;

    const offset = (parseInt(page as string) - 1) * parseInt(limit as string);
    let query = `SELECT * FROM customers WHERE is_active = true`;
    let countQuery = `SELECT COUNT(*) FROM customers WHERE is_active = true`;
    const params: any[] = [];
    let paramIndex = 1;

    if (search) {
      const searchCondition = ` AND (company_name ILIKE $${paramIndex} OR first_name ILIKE $${paramIndex} OR last_name ILIKE $${paramIndex} OR email ILIKE $${paramIndex} OR phone ILIKE $${paramIndex} OR customer_number ILIKE $${paramIndex})`;
      query += searchCondition;
      countQuery += searchCondition;
      params.push(`%${search}%`);
      paramIndex++;
    }

    if (status) {
      query += ` AND lifecycle_stage = $${paramIndex}`;
      countQuery += ` AND lifecycle_stage = $${paramIndex}`;
      params.push(status);
      paramIndex++;
    }

    if (customer_group) {
      query += ` AND customer_group = $${paramIndex}`;
      countQuery += ` AND customer_group = $${paramIndex}`;
      params.push(customer_group);
      paramIndex++;
    }

    const allowedSorts = ['created_at', 'company_name', 'customer_number', 'current_balance'];
    const sortField = allowedSorts.includes(sort as string) ? sort : 'created_at';
    const sortOrder = order === 'asc' ? 'ASC' : 'DESC';

    query += ` ORDER BY ${sortField} ${sortOrder} LIMIT $${paramIndex} OFFSET $${paramIndex + 1}`;
    params.push(parseInt(limit as string), offset);

    const [results, countResult] = await Promise.all([
      db.query(query, params),
      db.query(countQuery, params.slice(0, paramIndex - 1))
    ]);

    const total = parseInt(countResult.rows[0].count);

    res.json({
      success: true,
      data: results.rows,
      pagination: {
        page: parseInt(page as string),
        limit: parseInt(limit as string),
        total,
        pages: Math.ceil(total / parseInt(limit as string))
      }
    });
  } catch (error: any) {
    console.error('Error fetching customers:', error);
    res.status(500).json({ success: false, error: 'שגיאה בטעינת לקוחות', details: error.message });
  }
});

// GET /api/customers/:id
router.get('/:id', async (req: Request, res: Response) => {
  try {
    const result = await db.query('SELECT * FROM customers WHERE id = $1 AND is_active = true', [req.params.id]);
    if (result.rows.length === 0) {
      return res.status(404).json({ success: false, error: 'לקוח לא נמצא' });
    }
    res.json({ success: true, data: result.rows[0] });
  } catch (error: any) {
    res.status(500).json({ success: false, error: 'שגיאה', details: error.message });
  }
});

// POST /api/customers
router.post('/', async (req: Request, res: Response) => {
  try {
    const { company_name, customer_type, email, phone, ...rest } = req.body;

    if (!company_name && !rest.first_name) {
      return res.status(400).json({ success: false, error: 'שם חברה או שם פרטי חובה' });
    }

    // Auto-generate customer number
    const lastCustomer = await db.query("SELECT customer_number FROM customers ORDER BY created_at DESC LIMIT 1");
    let nextNum = 1;
    if (lastCustomer.rows.length > 0) {
      const match = lastCustomer.rows[0].customer_number?.match(/(\d+)$/);
      if (match) nextNum = parseInt(match[1]) + 1;
    }
    const customer_number = `C-2026-${String(nextNum).padStart(3, '0')}`;

    const fields = { customer_number, company_name, customer_type, email, phone, ...rest };
    const keys = Object.keys(fields).filter(k => fields[k] !== undefined);
    const values = keys.map(k => fields[k]);
    const placeholders = keys.map((_, i) => `$${i + 1}`);

    const result = await db.query(
      `INSERT INTO customers (${keys.join(',')}) VALUES (${placeholders.join(',')}) RETURNING *`,
      values
    );

    // Audit log
    await db.query(
      `INSERT INTO audit_log (action, entity_type, entity_id, new_values) VALUES ('create', 'customer', $1, $2)`,
      [result.rows[0].id, JSON.stringify(result.rows[0])]
    );

    res.status(201).json({ success: true, data: result.rows[0] });
  } catch (error: any) {
    res.status(500).json({ success: false, error: 'שגיאה ביצירת לקוח', details: error.message });
  }
});

// PUT /api/customers/:id
router.put('/:id', async (req: Request, res: Response) => {
  try {
    const old = await db.query('SELECT * FROM customers WHERE id = $1', [req.params.id]);
    if (old.rows.length === 0) {
      return res.status(404).json({ success: false, error: 'לקוח לא נמצא' });
    }

    const fields = req.body;
    fields.updated_at = new Date();
    const keys = Object.keys(fields).filter(k => fields[k] !== undefined);
    const values = keys.map(k => fields[k]);
    const setClause = keys.map((k, i) => `${k} = $${i + 1}`).join(', ');

    const result = await db.query(
      `UPDATE customers SET ${setClause} WHERE id = $${keys.length + 1} RETURNING *`,
      [...values, req.params.id]
    );

    // Audit log
    await db.query(
      `INSERT INTO audit_log (action, entity_type, entity_id, old_values, new_values) VALUES ('update', 'customer', $1, $2, $3)`,
      [req.params.id, JSON.stringify(old.rows[0]), JSON.stringify(result.rows[0])]
    );

    res.json({ success: true, data: result.rows[0] });
  } catch (error: any) {
    res.status(500).json({ success: false, error: 'שגיאה בעדכון', details: error.message });
  }
});

// DELETE /api/customers/:id (soft delete)
router.delete('/:id', async (req: Request, res: Response) => {
  try {
    const result = await db.query(
      'UPDATE customers SET is_active = false, updated_at = NOW() WHERE id = $1 RETURNING *',
      [req.params.id]
    );
    if (result.rows.length === 0) {
      return res.status(404).json({ success: false, error: 'לקוח לא נמצא' });
    }

    await db.query(
      `INSERT INTO audit_log (action, entity_type, entity_id, old_values) VALUES ('delete', 'customer', $1, $2)`,
      [req.params.id, JSON.stringify(result.rows[0])]
    );

    res.json({ success: true, message: 'לקוח נמחק בהצלחה' });
  } catch (error: any) {
    res.status(500).json({ success: false, error: 'שגיאה במחיקה', details: error.message });
  }
});

// GET /api/customers/stats
router.get('/stats', async (req: Request, res: Response) => {
  try {
    const stats = await db.query(`
      SELECT
        COUNT(*) FILTER (WHERE is_active = true) as total_active,
        COUNT(*) FILTER (WHERE lifecycle_stage = 'active') as active,
        COUNT(*) FILTER (WHERE lifecycle_stage = 'lead') as leads,
        COUNT(*) FILTER (WHERE customer_group = 'vip') as vip,
        COALESCE(SUM(current_balance) FILTER (WHERE is_active = true), 0) as total_balance
      FROM customers
    `);
    res.json({ success: true, data: stats.rows[0] });
  } catch (error: any) {
    res.status(500).json({ success: false, error: 'שגיאה', details: error.message });
  }
});

export default router;

2. **הוסף Auto-numbering trigger:**
```sql
CREATE OR REPLACE FUNCTION generate_next_number(prefix TEXT, table_name TEXT, column_name TEXT)
RETURNS TEXT AS $$
DECLARE
  last_num INTEGER;
  next_num TEXT;
BEGIN
  EXECUTE format('SELECT COALESCE(MAX(CAST(SUBSTRING(%I FROM ''[0-9]+$'') AS INTEGER)), 0) + 1 FROM %I WHERE %I IS NOT NULL', column_name, table_name, column_name) INTO last_num;
  next_num := prefix || LPAD(last_num::TEXT, 3, '0');
  RETURN next_num;
END;
$$ LANGUAGE plpgsql;
```

3. **הוסף Updated_at trigger:**
```sql
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply to all tables
DO $$
DECLARE
  t TEXT;
BEGIN
  FOR t IN
    SELECT table_name FROM information_schema.tables
    WHERE table_schema = 'public'
    AND table_type = 'BASE TABLE'
    AND EXISTS (
      SELECT 1 FROM information_schema.columns
      WHERE table_name = tables.table_name AND column_name = 'updated_at'
    )
  LOOP
    EXECUTE format('DROP TRIGGER IF EXISTS trg_updated_at ON %I', t);
    EXECUTE format('CREATE TRIGGER trg_updated_at BEFORE UPDATE ON %I FOR EACH ROW EXECUTE FUNCTION update_updated_at()', t);
  END LOOP;
END;
$$;
```
