import { TerminalTool } from "./terminalTool";
import { FileTool } from "./fileTool";
import { callLLM } from "../llm/client";
import { extractTextContent } from "../llm/parser";
import * as fs from "fs";
import * as path from "path";

export interface TestResult {
  passed: number;
  failed: number;
  skipped: number;
  total: number;
  duration: number;
  details: Array<{
    name: string;
    status: "pass" | "fail" | "skip";
    error?: string;
    duration: number;
  }>;
  output: string;
}

export class TestTool {
  private terminal: TerminalTool;
  private fileTool: FileTool;
  private workspaceDir: string;

  constructor(workspaceDir: string, terminal: TerminalTool, fileTool: FileTool) {
    this.workspaceDir = workspaceDir;
    this.terminal = terminal;
    this.fileTool = fileTool;
  }

  private detectTestFramework(): {
    framework: string;
    command: string;
    configFile?: string;
  } {
    const hasFile = (f: string) => fs.existsSync(path.join(this.workspaceDir, f));

    try {
      const pkg = JSON.parse(fs.readFileSync(path.join(this.workspaceDir, "package.json"), "utf-8"));
      const allDeps = { ...pkg.dependencies, ...pkg.devDependencies };

      if (allDeps.vitest) return { framework: "vitest", command: "npx vitest run --reporter=json" };
      if (allDeps.jest) return { framework: "jest", command: "npx jest --json --forceExit" };
      if (allDeps.mocha) return { framework: "mocha", command: "npx mocha --reporter json" };
      if (allDeps.playwright || allDeps["@playwright/test"])
        return { framework: "playwright", command: "npx playwright test --reporter=json" };
      if (allDeps.cypress)
        return { framework: "cypress", command: "npx cypress run --reporter json" };
    } catch {}

    if (hasFile("pytest.ini") || hasFile("conftest.py") || hasFile("setup.cfg"))
      return { framework: "pytest", command: "python -m pytest --tb=short -v" };
    if (hasFile("requirements.txt")) {
      const req = fs.readFileSync(path.join(this.workspaceDir, "requirements.txt"), "utf-8");
      if (req.includes("pytest")) return { framework: "pytest", command: "python -m pytest --tb=short -v" };
    }

    if (hasFile("go.mod")) return { framework: "go-test", command: "go test ./... -v -json" };
    if (hasFile("Cargo.toml")) return { framework: "cargo-test", command: "cargo test -- --format=json" };

    return { framework: "jest", command: "npx jest --json --forceExit" };
  }

  async runTests(options?: {
    pattern?: string;
    watch?: boolean;
    coverage?: boolean;
    verbose?: boolean;
  }): Promise<TestResult> {
    const detected = this.detectTestFramework();
    let command = detected.command;

    if (options?.pattern) {
      switch (detected.framework) {
        case "vitest": command += ` ${options.pattern}`; break;
        case "jest": command += ` --testPathPattern="${options.pattern}"`; break;
        case "pytest": command += ` -k "${options.pattern}"`; break;
        case "go-test": command = `go test ./${options.pattern}/... -v -json`; break;
      }
    }

    if (options?.coverage) {
      switch (detected.framework) {
        case "vitest": command += " --coverage"; break;
        case "jest": command += " --coverage"; break;
        case "pytest": command += " --cov"; break;
      }
    }

    const startTime = Date.now();
    const result = await this.terminal.execute(command, { timeout: 120000 });
    const duration = Date.now() - startTime;

    return this.parseTestOutput(detected.framework, result.stdout + "\n" + result.stderr, duration);
  }

  async runSingleTest(filePath: string): Promise<TestResult> {
    const detected = this.detectTestFramework();
    let command: string;

    switch (detected.framework) {
      case "vitest": command = `npx vitest run ${filePath} --reporter=json`; break;
      case "jest": command = `npx jest ${filePath} --json --forceExit`; break;
      case "pytest": command = `python -m pytest ${filePath} -v`; break;
      case "go-test": command = `go test -run ${filePath} -v -json`; break;
      default: command = `npx jest ${filePath} --json --forceExit`; break;
    }

    const startTime = Date.now();
    const result = await this.terminal.execute(command, { timeout: 60000 });
    return this.parseTestOutput(detected.framework, result.stdout + "\n" + result.stderr, Date.now() - startTime);
  }

  async generateTests(sourceFile: string): Promise<string> {
    const content = this.fileTool.readFile(sourceFile);
    if (!content.success) throw new Error(`Cannot read: ${sourceFile}`);

    const ext = path.extname(sourceFile);
    const basename = path.basename(sourceFile, ext);
    const testFile = sourceFile.replace(ext, `.test${ext}`);
    const detected = this.detectTestFramework();

    const response = await callLLM({
      system: `You are an expert test engineer. Generate comprehensive tests using ${detected.framework}.
Write tests that cover:
- Happy path for all exported functions/methods
- Edge cases and error handling
- Input validation
- Return value assertions
Respond with ONLY the test file code, no explanation.`,
      messages: [
        {
          role: "user",
          content: `Generate tests for this file (${sourceFile}):\n\n\`\`\`\n${content.output}\n\`\`\``,
        },
      ],
    });

    let testCode = extractTextContent(response.content);
    testCode = testCode.replace(/^```\w*\n/, "").replace(/\n```$/, "").trim();

    this.fileTool.createFile(testFile, testCode);
    return testFile;
  }

  async setupTestFramework(framework: "vitest" | "jest" | "playwright" | "pytest"): Promise<void> {
    switch (framework) {
      case "vitest": {
        await this.terminal.execute("npm install -D vitest @vitest/coverage-v8", { timeout: 60000 });
        this.fileTool.createFile("vitest.config.ts", `import { defineConfig } from 'vitest/config';

export default defineConfig({
  test: {
    globals: true,
    environment: 'node',
    coverage: {
      reporter: ['text', 'json', 'html'],
    },
  },
});`);
        break;
      }
      case "jest": {
        await this.terminal.execute("npm install -D jest ts-jest @types/jest", { timeout: 60000 });
        this.fileTool.createFile("jest.config.js", `module.exports = {
  preset: 'ts-jest',
  testEnvironment: 'node',
  roots: ['<rootDir>/src'],
  testMatch: ['**/__tests__/**/*.[jt]s', '**/?(*.)+(spec|test).[jt]s'],
  collectCoverageFrom: ['src/**/*.{js,ts}', '!src/**/*.d.ts'],
};`);
        break;
      }
      case "playwright": {
        await this.terminal.execute("npm install -D @playwright/test && npx playwright install", { timeout: 180000 });
        this.fileTool.createFile("playwright.config.ts", `import { defineConfig } from '@playwright/test';

export default defineConfig({
  testDir: './tests',
  fullyParallel: true,
  retries: 2,
  use: {
    baseURL: 'http://localhost:3000',
    trace: 'on-first-retry',
  },
  webServer: {
    command: 'npm run dev',
    port: 3000,
    reuseExistingServer: true,
  },
});`);
        break;
      }
      case "pytest": {
        await this.terminal.execute("pip install pytest pytest-cov pytest-asyncio", { timeout: 60000 });
        this.fileTool.createFile("pytest.ini", `[pytest]
testpaths = tests
python_files = test_*.py
python_classes = Test*
python_functions = test_*
addopts = -v --tb=short`);
        break;
      }
    }
  }

  private parseTestOutput(framework: string, output: string, duration: number): TestResult {
    const result: TestResult = {
      passed: 0,
      failed: 0,
      skipped: 0,
      total: 0,
      duration,
      details: [],
      output,
    };

    try {
      if (framework === "jest" || framework === "vitest") {
        const jsonMatch = output.match(/\{[\s\S]*"numPassedTests"[\s\S]*\}/);
        if (jsonMatch) {
          const data = JSON.parse(jsonMatch[0]);
          result.passed = data.numPassedTests || 0;
          result.failed = data.numFailedTests || 0;
          result.skipped = data.numPendingTests || 0;
          result.total = data.numTotalTests || 0;

          if (data.testResults) {
            for (const suite of data.testResults) {
              for (const test of suite.assertionResults || []) {
                result.details.push({
                  name: test.fullName || test.title,
                  status: test.status === "passed" ? "pass" : test.status === "failed" ? "fail" : "skip",
                  error: test.failureMessages?.join("\n"),
                  duration: test.duration || 0,
                });
              }
            }
          }
          return result;
        }
      }
    } catch {}

    // Fallback: parse text output
    const passMatch = output.match(/(\d+)\s+pass/i);
    const failMatch = output.match(/(\d+)\s+fail/i);
    const skipMatch = output.match(/(\d+)\s+skip/i);

    result.passed = passMatch ? parseInt(passMatch[1]) : 0;
    result.failed = failMatch ? parseInt(failMatch[1]) : 0;
    result.skipped = skipMatch ? parseInt(skipMatch[1]) : 0;
    result.total = result.passed + result.failed + result.skipped;

    return result;
  }
}
