// Add imports:
import { BuildFlow } from "./flows/buildFlow";
import { DiagnosticFlow } from "./flows/diagnosticFlow";
import { DataFlow } from "./flows/dataFlow";
import { RealtimeFlow } from "./flows/realtimeFlow";
import { QualityGateFlow } from "./flows/qualityGateFlow";
import { UpgradeFlow } from "./flows/upgradeFlow";

// After server creation:
const buildFlow = new BuildFlow({
  workspaceDir: WORKSPACE_DIR,
  maxRetries: 3,
  autoTest: true,
  autoLint: true,
  autoSnapshot: true,
  autoPreview: true,
}, (msg) => console.log(msg), (phase, status) => {
  wsServer.broadcast({ type: "phase", taskId: "build", data: { phase, status }, timestamp: new Date() });
});

const diagnosticFlow = new DiagnosticFlow(WORKSPACE_DIR, (msg) => console.log(msg));
const dataFlow = new DataFlow(WORKSPACE_DIR, (msg) => console.log(msg));
const realtimeFlow = new RealtimeFlow(server, WORKSPACE_DIR, terminal, (msg) => console.log(msg));
const qualityGate = new QualityGateFlow(WORKSPACE_DIR, (msg) => console.log(msg));
const upgradeFlow = new UpgradeFlow(WORKSPACE_DIR, (msg) => console.log(msg));

// ============ FLOW ROUTES ============

// Full build flow
app.post("/api/flow/build", async (req, res) => {
  const { task, context } = req.body;
  const result = await buildFlow.run(task, context);
  res.json(result);
});

// Diagnose and fix
app.post("/api/flow/diagnose", async (req, res) => {
  const { problem } = req.body;
  const result = await diagnosticFlow.diagnoseAndFix(problem);
  res.json(result);
});

// Quick fix
app.post("/api/flow/quickfix", async (req, res) => {
  const { error } = req.body;
  const result = await diagnosticFlow.quickFix(error);
  res.json(result);
});

// Server recovery
app.post("/api/flow/recover-server", async (_, res) => {
  const result = await diagnosticFlow.recoverServer();
  res.json(result);
});

// Create model
app.post("/api/flow/model", async (req, res) => {
  const { description } = req.body;
  const result = await dataFlow.createModel(description);
  res.json({ model: result.model, files: Object.fromEntries(result.files) });
});

// Add field
app.post("/api/flow/model/:name/field", async (req, res) => {
  const migration = await dataFlow.addField(req.params.name, req.body);
  res.json({ migration });
});

// Seed data
app.post("/api/flow/model/:name/seed", async (req, res) => {
  const file = await dataFlow.seedModel(req.params.name, req.body.count);
  res.json({ file });
});

// Data cleanup
app.post("/api/flow/model/:name/cleanup", async (req, res) => {
  const result = await dataFlow.cleanupData(req.params.name, req.body);
  res.json(result);
});

// Quality gate
app.post("/api/flow/quality-gate", async (req, res) => {
  const result = await qualityGate.runAllGates(req.body);
  res.json(result);
});

// Upgrade dependencies
app.post("/api/flow/upgrade", async (req, res) => {
  const result = await upgradeFlow.upgradeAll(req.body);
  res.json(result);
});

// Real-time channels
app.get("/api/realtime/channels", (_, res) => {
  res.json({ channels: realtimeFlow.getChannels() });
});

app.get("/api/realtime/history/:channel", (req, res) => {
  res.json({ events: realtimeFlow.getEventHistory(req.params.channel) });
});
