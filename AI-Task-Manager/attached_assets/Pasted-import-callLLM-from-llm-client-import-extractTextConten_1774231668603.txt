import { callLLM } from "../llm/client";
import { extractTextContent } from "../llm/parser";
import { FileTool } from "./fileTool";
import { TerminalTool } from "./terminalTool";

export class EmailTool {
  private fileTool: FileTool;
  private terminal: TerminalTool;

  constructor(fileTool: FileTool, terminal: TerminalTool) {
    this.fileTool = fileTool;
    this.terminal = terminal;
  }

  async setupEmailService(provider: "nodemailer" | "resend" | "sendgrid"): Promise<void> {
    const packages: Record<string, string> = {
      nodemailer: "nodemailer @types/nodemailer",
      resend: "resend",
      sendgrid: "@sendgrid/mail",
    };

    await this.terminal.execute(`npm install ${packages[provider]}`, { timeout: 30000 });

    const serviceCode: Record<string, string> = {
      nodemailer: `import nodemailer from 'nodemailer';

const transporter = nodemailer.createTransport({
  host: process.env.SMTP_HOST || 'smtp.gmail.com',
  port: parseInt(process.env.SMTP_PORT || '587'),
  secure: false,
  auth: {
    user: process.env.SMTP_USER,
    pass: process.env.SMTP_PASS,
  },
});

export async function sendEmail(params: {
  to: string | string[];
  subject: string;
  html: string;
  text?: string;
  from?: string;
  replyTo?: string;
  attachments?: Array<{ filename: string; content: Buffer | string }>;
}) {
  return transporter.sendMail({
    from: params.from || process.env.EMAIL_FROM || 'noreply@app.com',
    to: Array.isArray(params.to) ? params.to.join(', ') : params.to,
    subject: params.subject,
    html: params.html,
    text: params.text,
    replyTo: params.replyTo,
    attachments: params.attachments,
  });
}`,

      resend: `import { Resend } from 'resend';

const resend = new Resend(process.env.RESEND_API_KEY);

export async function sendEmail(params: {
  to: string | string[];
  subject: string;
  html: string;
  from?: string;
}) {
  return resend.emails.send({
    from: params.from || process.env.EMAIL_FROM || 'noreply@app.com',
    to: params.to,
    subject: params.subject,
    html: params.html,
  });
}`,

      sendgrid: `import sgMail from '@sendgrid/mail';

sgMail.setApiKey(process.env.SENDGRID_API_KEY!);

export async function sendEmail(params: {
  to: string | string[];
  subject: string;
  html: string;
  from?: string;
}) {
  return sgMail.send({
    from: params.from || process.env.EMAIL_FROM || 'noreply@app.com',
    to: params.to,
    subject: params.subject,
    html: params.html,
  });
}`,
    };

    this.fileTool.createFile("src/services/email.ts", serviceCode[provider]);
  }

  async generateTemplate(
    name: string,
    type: "welcome" | "reset-password" | "verification" | "invoice" | "notification" | "custom",
    options?: { brandColor?: string; logoUrl?: string; companyName?: string }
  ): Promise<string> {
    const brandColor = options?.brandColor || "#6c5ce7";
    const companyName = options?.companyName || "App";

    const response = await callLLM({
      system: `Generate a professional, responsive HTML email template.
Rules:
- Use tables for layout (email client compatibility)
- Inline all CSS styles
- Max width 600px, centered
- Support dark mode with media queries
- Include preheader text
- Mobile responsive
- Brand color: ${brandColor}
- Company: ${companyName}
- Include unsubscribe link placeholder
- Use {{variable}} syntax for dynamic content

Respond with ONLY the HTML.`,
      messages: [{
        role: "user",
        content: `Generate a "${type}" email template named "${name}".
${type === "welcome" ? "Include: greeting, feature highlights, CTA button, social links" : ""}
${type === "reset-password" ? "Include: reset link button, security notice, expiry warning" : ""}
${type === "verification" ? "Include: verification code/link, expiry time" : ""}
${type === "invoice" ? "Include: invoice number, line items table, total, payment link" : ""}
${type === "notification" ? "Include: notification title, message body, action button" : ""}`,
      }],
      maxTokens: 4096,
    });

    let html = extractTextContent(response.content);
    html = html.replace(/^```\w*\n/, "").replace(/\n```$/, "").trim();

    this.fileTool.createFile(`src/emails/templates/${name}.html`, html);

    // Generate render function
    const renderFunc = `export function render${name.charAt(0).toUpperCase() + name.slice(1)}(data: Record<string, string>): string {
  let html = require('fs').readFileSync(__dirname + '/templates/${name}.html', 'utf-8');
  for (const [key, value] of Object.entries(data)) {
    html = html.replace(new RegExp('\\\\{\\\\{' + key + '\\\\}\\\\}', 'g'), value);
  }
  return html;
}
`;

    // Append to emails index
    const indexPath = "src/emails/index.ts";
    const existing = this.fileTool.readFile(indexPath);
    const content = existing.success ? existing.output + "\n" + renderFunc : renderFunc;
    this.fileTool.createFile(indexPath, content);

    return html;
  }

  async previewTemplate(templatePath: string, data: Record<string, string>): Promise<string> {
    const content = this.fileTool.readFile(templatePath);
    if (!content.success) throw new Error(`Cannot read: ${templatePath}`);

    let html = content.output || "";
    for (const [key, value] of Object.entries(data)) {
      html = html.replace(new RegExp(`\\{\\{${key}\\}\\}`, "g"), value);
    }

    const previewPath = "/tmp/email-preview.html";
    this.fileTool.createFile(previewPath, html);
    return previewPath;
  }
}
