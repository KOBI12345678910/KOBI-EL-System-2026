  async scaffold(templateName: string): Promise<{ success: boolean; output: string }> {
    const template = this.getTemplates()[templateName];
    if (!template) {
      return { success: false, output: `Template '${templateName}' not found. Available: ${Object.keys(this.getTemplates()).join(", ")}` };
    }

    let output = `Scaffolding: ${template.name}\n`;

    // Run init commands
    if (template.commands) {
      for (const cmd of template.commands) {
        output += `\n> ${cmd}\n`;
        const result = await this.terminal.execute(cmd, { timeout: 300000 });
        output += result.stdout + "\n";
        if (!result.success) {
          output += `Warning: ${result.stderr}\n`;
        }
      }
    }

    // Install packages
    if (template.packages?.length) {
      output += `\nInstalling: ${template.packages.join(", ")}\n`;
      await this.packageTool.install(template.packages);
    }
    if (template.devPackages?.length) {
      output += `Installing dev: ${template.devPackages.join(", ")}\n`;
      await this.packageTool.install(template.devPackages, { dev: true });
    }

    // Create files (overwrite if template provides them)
    for (const [filePath, content] of Object.entries(template.files)) {
      this.fileTool.createFile(filePath, content);
      output += `Created: ${filePath}\n`;
    }

    return { success: true, output };
  }
