// ═══════════════════════════════════════════
// 1. LLM CACHE — לא שולח אותה שאלה פעמיים
// ═══════════════════════════════════════════

interface CacheEntry {
  response: any;
  timestamp: number;
  tokens: number;
  hits: number;
}

class LLMCache {
  private cache: Map<string, CacheEntry> = new Map();
  private maxSize = 500;
  private ttl = 30 * 60 * 1000; // 30 דקות

  private makeKey(system: string, messages: any[]): string {
    const raw = system + JSON.stringify(messages);
    // Simple hash
    let hash = 0;
    for (let i = 0; i < raw.length; i++) {
      const char = raw.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash |= 0;
    }
    return `llm_${Math.abs(hash).toString(36)}`;
  }

  get(system: string, messages: any[]): any | null {
    const key = this.makeKey(system, messages);
    const entry = this.cache.get(key);
    if (!entry) return null;
    if (Date.now() - entry.timestamp > this.ttl) {
      this.cache.delete(key);
      return null;
    }
    entry.hits++;
    return entry.response;
  }

  set(system: string, messages: any[], response: any, tokens: number): void {
    const key = this.makeKey(system, messages);

    // Evict old entries if full
    if (this.cache.size >= this.maxSize) {
      let oldestKey = "";
      let oldestTime = Infinity;
      for (const [k, v] of this.cache) {
        if (v.timestamp < oldestTime) { oldestTime = v.timestamp; oldestKey = k; }
      }
      if (oldestKey) this.cache.delete(oldestKey);
    }

    this.cache.set(key, { response, timestamp: Date.now(), tokens, hits: 0 });
  }

  getStats(): { size: number; hits: number; savedTokens: number } {
    let hits = 0, savedTokens = 0;
    for (const [, entry] of this.cache) {
      hits += entry.hits;
      savedTokens += entry.hits * entry.tokens;
    }
    return { size: this.cache.size, hits, savedTokens };
  }

  clear(): void { this.cache.clear(); }
}

// ═══════════════════════════════════════════
// 2. CONNECTION POOL — חיבורים מוכנים מראש
// ═══════════════════════════════════════════

class AnthropicPool {
  private clients: Anthropic[] = [];
  private currentIndex = 0;
  private poolSize: number;

  constructor(poolSize = 3) {
    this.poolSize = poolSize;
    for (let i = 0; i < poolSize; i++) {
      this.clients.push(new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY! }));
    }
  }

  getClient(): Anthropic {
    const client = this.clients[this.currentIndex];
    this.currentIndex = (this.currentIndex + 1) % this.poolSize;
    return client;
  }
}

// 3. PARALLEL EXECUTOR — הרצה מקבילית
// ═══════════════════════════════════════════

export class ParallelExecutor {
  private maxConcurrency: number;
  private running = 0;
  private queue: Array<{ fn: () => Promise<any>; resolve: (v: any) => void; reject: (e: any) => void }> = [];

  constructor(maxConcurrency = 5) {
    this.maxConcurrency = maxConcurrency;
  }

  async run<T>(fn: () => Promise<T>): Promise<T> {
    return new Promise((resolve, reject) => {
      this.queue.push({ fn, resolve, reject });
      this.processQueue();
    });
  }

  async runAll<T>(fns: Array<() => Promise<T>>): Promise<T[]> {
    return Promise.all(fns.map((fn) => this.run(fn)));
  }

  async runBatch<T, R>(items: T[], fn: (item: T) => Promise<R>): Promise<R[]> {
    return this.runAll(items.map((item) => () => fn(item)));
  }

  private async processQueue(): Promise<void> {
    while (this.queue.length > 0 && this.running < this.maxConcurrency) {
      const task = this.queue.shift();
      if (!task) break;

      this.running++;
      task.fn()
        .then(task.resolve)
        .catch(task.reject)
        .finally(() => {
          this.running--;
          this.processQueue();
        });
    }
  }
}

// 4. FAST LLM — API מהיר עם cache + stream + retry
// ═══════════════════════════════════════════

const llmCache = new LLMCache();
const pool = new AnthropicPool(3);

export interface FastLLMOptions {
  system: string;
  messages: Array<{ role: "user" | "assistant"; content: any }>;
  model?: string;
  maxTokens?: number;
  tools?: any[];
  temperature?: number;
  cache?: boolean;
  stream?: boolean;
  timeout?: number;
  onText?: (text: string) => void;
  onToolUse?: (tool: { id: string; name: string; input: any }) => void;
}

export async function fastLLM(options: FastLLMOptions): Promise<{
  content: any[];
  stopReason: string;
  inputTokens: number;
  outputTokens: number;
  cached: boolean;
  duration: number;
}> {
  const startTime = Date.now();
  const model = options.model || process.env.MODEL || "claude-sonnet-4-20250514";
  const maxTokens = options.maxTokens || 8192;

  // Check cache first
  if (options.cache !== false) {
    const cached = llmCache.get(options.system, options.messages);
    if (cached) {
      return {
        ...cached,
        cached: true,
        duration: Date.now() - startTime,
      };
    }
  }

  const client = pool.getClient();
  let attempts = 0;
  const maxAttempts = 3;
  let lastError: any;

  while (attempts < maxAttempts) {
    attempts++;

    try {
      if (options.stream && (options.onText || options.onToolUse)) {
        // ═══ STREAMING MODE — תשובות מתחילות להגיע מיד ═══
        const stream = client.messages.stream({
          model,
          max_tokens: maxTokens,
          system: options.system,
          messages: options.messages as any,
          tools: options.tools as any,
          temperature: options.temperature,
        });

        let currentToolId = "";
        let currentToolName = "";
        let toolInputJson = "";
        const content: any[] = [];
        let fullText = "";

        for await (const event of stream) {
          if (event.type === "content_block_start") {
            if (event.content_block.type === "text") {
              content.push({ type: "text", text: "" });
            } else if (event.content_block.type === "tool_use") {
              currentToolId = event.content_block.id;
              currentToolName = event.content_block.name;
              toolInputJson = "";
            }
          }
          if (event.type === "content_block_delta") {
            if (event.delta.type === "text_delta") {
              const text = event.delta.text;
              fullText += text;
              if (content.length > 0 && content[content.length - 1].type === "text") {
                content[content.length - 1].text += text;
              }
              options.onText?.(text);
            }
            if (event.delta.type === "input_json_delta") {
              toolInputJson += event.delta.partial_json;
            }
          }
          if (event.type === "content_block_stop" && currentToolName) {
            try {
              const input = JSON.parse(toolInputJson);
              content.push({ type: "tool_use", id: currentToolId, name: currentToolName, input });
              options.onToolUse?.({ id: currentToolId, name: currentToolName, input });
            } catch {}
            currentToolName = "";
            toolInputJson = "";
          }
        }

        const final = await stream.finalMessage();
        const result = {
          content,
          stopReason: final.stop_reason || "end_turn",
          inputTokens: final.usage.input_tokens,
          outputTokens: final.usage.output_tokens,
          cached: false,
          duration: Date.now() - startTime,
        };

        // Cache the result
        if (options.cache !== false) {
          llmCache.set(options.system, options.messages, result, final.usage.input_tokens + final.usage.output_tokens);
        }

        return result;
      } else {
        // ═══ NON-STREAMING — מהיר לתשובות קצרות ═══
        const response = await client.messages.create({
          model,
          max_tokens: maxTokens,
          system: options.system,
          messages: options.messages as any,
          tools: options.tools as any,
          temperature: options.temperature,
        });

        const result = {
          content: response.content as any[],
          stopReason: response.stop_reason || "end_turn",
          inputTokens: response.usage.input_tokens,
          outputTokens: response.usage.output_tokens,
          cached: false,
          duration: Date.now() - startTime,
        };

        if (options.cache !== false) {
          llmCache.set(options.system, options.messages, result, response.usage.input_tokens + response.usage.output_tokens);
        }

        return result;
      }
    } catch (err: any) {
      lastError = err;

      // Retry on rate limit or server error
      if (err.status === 429 || err.status >= 500) {
        const delay = Math.min(1000 * Math.pow(2, attempts), 10000);
        await new Promise((r) => setTimeout(r, delay));
        continue;
      }

      throw err;
    }
  }

  throw lastError || new Error("Max retries exceeded");
}

// 5. FAST COMMAND — הרצת פקודות מהירה
// ═══════════════════════════════════════════

import { spawn } from "child_process";
import treeKill from "tree-kill";

const commandCache = new Map<string, { output: string; timestamp: number }>();
const COMMAND_CACHE_TTL = 5000; // 5 שניות

export async function fastExec(
  command: string,
  options?: {
    cwd?: string;
    timeout?: number;
    cache?: boolean;
    onOutput?: (data: string) => void;
  }
): Promise<{ success: boolean; stdout: string; stderr: string; duration: number }> {
  const startTime = Date.now();

  // Cache for repeated commands (like `ls`, `cat`, etc.)
  if (options?.cache !== false) {
    const cached = commandCache.get(command);
    if (cached && Date.now() - cached.timestamp < COMMAND_CACHE_TTL) {
      return { success: true, stdout: cached.output, stderr: "", duration: 0 };
    }
  }

  return new Promise((resolve) => {
    let stdout = "";
    let stderr = "";
    let killed = false;

    const proc = spawn("bash", ["-c", command], {
      cwd: options?.cwd || process.env.WORKSPACE_DIR || "./workspace",
      env: process.env,
      stdio: ["pipe", "pipe", "pipe"],
    });

    const timer = setTimeout(() => {
      killed = true;
      if (proc.pid) treeKill(proc.pid, "SIGKILL");
    }, options?.timeout || 30000);

    proc.stdout?.on("data", (data: Buffer) => {
      const text = data.toString();
      stdout += text;
      options?.onOutput?.(text);
    });

    proc.stderr?.on("data", (data: Buffer) => {
      stderr += data.toString();
    });

    proc.on("close", (code) => {
      clearTimeout(timer);
      const result = {
        success: code === 0 && !killed,
        stdout: stdout.trim(),
        stderr: stderr.trim(),
        duration: Date.now() - startTime,
      };

      // Cache successful reads
      if (result.success && options?.cache !== false) {
        commandCache.set(command, { output: result.stdout, timestamp: Date.now() });
      }

      resolve(result);
    });

    proc.on("error", (err) => {
      clearTimeout(timer);
      resolve({ success: false, stdout, stderr: err.message, duration: Date.now() - startTime });
    });
  });
}

// 6. BATCH OPERATIONS — פעולות בבת אחת
// ═══════════════════════════════════════════

export class BatchFileOps {
  static async readMultiple(filePaths: string[]): Promise<Map<string, string>> {
    const results = new Map<string, string>();
    const command = filePaths.map((f) => `echo "===FILE:${f}===" && cat "${f}" 2>/dev/null || echo "FILE_NOT_FOUND"`).join(" && ");

    const result = await fastExec(command, { timeout: 10000 });
    if (!result.success) return results;

    const parts = result.stdout.split(/===FILE:(.+?)===/);
    for (let i = 1; i < parts.length; i += 2) {
      const path = parts[i];
      const content = parts[i + 1]?.trim();
      if (content && content !== "FILE_NOT_FOUND") {
        results.set(path, content);
      }
    }

    return results;
  }

  static async writeMultiple(files: Array<{ path: string; content: string }>): Promise<number> {
    let written = 0;
    // Use a single shell script
    const commands = files.map((f) => {
      const dir = f.path.substring(0, f.path.lastIndexOf("/"));
      const escaped = f.content.replace(/'/g, "'\\''");
      return `mkdir -p "${dir}" && printf '%s' '${escaped}' > "${f.path}"`;
    });

    // Batch in groups of 5
    for (let i = 0; i < commands.length; i += 5) {
      const batch = commands.slice(i, i + 5).join(" && ");
      const result = await fastExec(batch, { timeout: 10000 });
      if (result.success) written += Math.min(5, commands.length - i);
    }

    return written;
  }

  static async existsMultiple(paths: string[]): Promise<Map<string, boolean>> {
    const results = new Map<string, boolean>();
    const command = paths.map((p) => `test -e "${p}" && echo "EXISTS:${p}" || echo "MISSING:${p}"`).join(" && ");
    const result = await fastExec(command, { timeout: 5000, cache: true });

    for (const line of result.stdout.split("\n")) {
      if (line.startsWith("EXISTS:")) results.set(line.slice(7), true);
      else if (line.startsWith("MISSING:")) results.set(line.slice(8), false);
    }

    return results;
  }
}

// 7. SMART STEP SKIPPING — דילוג על צעדים מיותרים
// ═══════════════════════════════════════════

export class StepOptimizer {
  private completedOperations: Set<string> = new Set();

  shouldSkip(step: { type: string; details: any }): boolean {
    const key = this.stepKey(step);

    // Skip if already done
    if (this.completedOperations.has(key)) return true;

    // Skip install if node_modules exists and package hasn't changed
    if (step.type === "install_package") {
      // Check if already installed
      return false; // Let the package tool handle dedup
    }

    // Skip create_file if file exists with same content
    if (step.type === "create_file" && step.details?.path) {
      // Will be checked by the executor
      return false;
    }

    return false;
  }

  markCompleted(step: { type: string; details: any }): void {
    this.completedOperations.add(this.stepKey(step));
  }

  private stepKey(step: { type: string; details: any }): string {
    return `${step.type}:${JSON.stringify(step.details || {})}`.slice(0, 200);
  }

  reset(): void {
    this.completedOperations.clear();
  }
}
