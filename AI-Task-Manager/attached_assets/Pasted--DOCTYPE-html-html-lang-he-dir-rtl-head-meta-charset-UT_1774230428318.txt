  <script>
    // ============ STATE ============
    let ws = null;
    let isRunning = false;
    let currentSteps = [];

    // ============ WEBSOCKET ============
    function connectWS() {
      const protocol = location.protocol === 'https:' ? 'wss:' : 'ws:';
      ws = new WebSocket(`${protocol}//${location.host}/ws`);

      ws.onopen = () => {
        document.getElementById('statusBadge').className = 'status-badge connected';
        document.getElementById('statusText').textContent = 'Connected';
      };

      ws.onclose = () => {
        document.getElementById('statusBadge').className = 'status-badge disconnected';
        document.getElementById('statusText').textContent = 'Disconnected';
        setTimeout(connectWS, 3000);
      };

      ws.onmessage = (e) => {
        const event = JSON.parse(e.data);
        handleEvent(event);
      };
    }

    // ============ EVENT HANDLER ============
    function handleEvent(event) {
      switch (event.type) {
        case 'plan':
          currentSteps = event.data.plan.steps || [];
          renderSteps();
          addMessage('agent', `📋 Plan: ${event.data.plan.taskSummary}\n${currentSteps.length} steps to execute`);
          break;

        case 'step_start':
          updateStepStatus(event.data.stepId, 'running');
          updateProgress(event.data.progress);
          break;

        case 'step_complete':
          updateStepStatus(event.data.stepId, 'success');
          updateProgress(event.data.progress);
          break;

        case 'step_failed':
          updateStepStatus(event.data.stepId, 'failed');
          break;

        case 'log':
          addLog(event.data.message);
          break;

        case 'status':
          addMessage('system', `[${event.data.status}] ${event.data.message}`);
          break;

        case 'error':
          addMessage('error', `❌ ${event.data.message}`);
          break;

        case 'complete':
          isRunning = false;
          updateSendButton();
          const icon = event.data.success ? '✅' : '⚠️';
          addMessage('agent', `${icon} ${event.data.summary}`);
          break;
      }
    }

    // ============ UI FUNCTIONS ============
    function addMessage(type, text) {
      const el = document.createElement('div');
      el.className = `message ${type}`;
      el.textContent = text;
      const container = document.getElementById('messages');
      container.appendChild(el);
      container.scrollTop = container.scrollHeight;
    }

    function addLog(text) {
      const el = document.createElement('div');
      el.className = 'log-line';
      if (text.includes('✅')) el.className += ' success';
      else if (text.includes('❌')) el.className += ' error';
      else if (text.includes('🔄') || text.includes('📦')) el.className += ' info';
      el.textContent = text;
      const panel = document.getElementById('logPanel');
      panel.appendChild(el);
      panel.scrollTop = panel.scrollHeight;
    }

    function renderSteps() {
      const list = document.getElementById('stepsList');
      list.innerHTML = currentSteps.map(step => `
        <div class="step-item" id="step-${step.id}">
          <div class="step-icon pending" id="step-icon-${step.id}">
            ${step.id}
          </div>
          <div class="step-info">
            <div>${step.description}</div>
            <div class="step-type">${step.type}</div>
          </div>
        </div>
      `).join('');
    }

    function updateStepStatus(stepId, status) {
      const icon = document.getElementById(`step-icon-${stepId}`);
      if (!icon) return;
      icon.className = `step-icon ${status}`;
      const symbols = { pending: stepId, running: '⟳', success: '✓', failed: '✗' };
      icon.textContent = symbols[status] || stepId;
    }

    function updateProgress(progressStr) {
      if (!progressStr) return;
      const [done, total] = progressStr.split('/').map(Number);
      const pct = Math.round((done / total) * 100);
      document.getElementById('progressFill').style.width = `${pct}%`;
      document.getElementById('progressText').textContent = `${done}/${total} steps (${pct}%)`;
    }

    function updateSendButton() {
      const btn = document.getElementById('sendBtn');
      if (isRunning) {
        btn.className = 'send-btn stop-btn';
        btn.textContent = '■';
        btn.onclick = stopTask;
      } else {
        btn.className = 'send-btn';
        btn.textContent = '▶';
        btn.onclick = sendTask;
      }
    }

    // ============ API CALLS ============
    async function sendTask() {
      const input = document.getElementById('taskInput');
      const task = input.value.trim();
      if (!task) return;

      addMessage('user', task);
      input.value = '';

      isRunning = true;
      updateSendButton();

      try {
        const res = await fetch('/api/task', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ task }),
        });
        const data = await res.json();
        if (!res.ok) {
          addMessage('error', `❌ ${data.error}`);
        }
      } catch (err) {
        addMessage('error', `❌ ${err.message}`);
      }

      isRunning = false;
      updateSendButton();
    }

    async function stopTask() {
      await fetch('/api/stop', { method: 'POST' });
      isRunning = false;
      updateSendButton();
      addMessage('system', 'Task stopped by user');
    }

    // ============ INPUT HANDLING ============
    document.getElementById('taskInput').addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) {
        e.preventDefault();
        if (isRunning) return;
        sendTask();
      }
    });

    // Auto-resize textarea
    document.getElementById('taskInput').addEventListener('input', function() {
      this.style.height = 'auto';
      this.style.height = Math.min(this.scrollHeight, 200) + 'px';
    });

    // ============ INIT ============
    connectWS();
  </script>
