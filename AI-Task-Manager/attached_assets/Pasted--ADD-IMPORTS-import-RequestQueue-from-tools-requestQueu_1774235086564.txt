// ═══ ADD IMPORTS ═══
import { RequestQueue } from "./tools/requestQueue";
import { CheckpointTool } from "./tools/checkpointTool";
import { AgentModeManager } from "./agent/agentModes";
import { PlanMode } from "./flows/planMode";
import { HotfixFlow } from "./flows/hotfixFlow";
import { HealthDashboardTool } from "./tools/healthDashboardTool";
import { RulesSyncTool } from "./tools/rulesSyncTool";
import { TokenTracker } from "./tools/tokenTracker";

// ═══ INITIALIZE ═══
const modeManager = new AgentModeManager();
const tokenTracker = new TokenTracker();
const rulesSyncTool = new RulesSyncTool(WORKSPACE_DIR, fileTool);
const checkpointTool = new CheckpointTool(WORKSPACE_DIR, terminal, (msg) => console.log(msg));
const hotfixFlow = new HotfixFlow(WORKSPACE_DIR, (msg) => console.log(msg));

const requestQueue = new RequestQueue(
  async (message, context) => agent.executeTask(message, context),
  (req) => wsServer.broadcast({ type: "queue_update", taskId: "system", data: req, timestamp: new Date() })
);

const planMode = new PlanMode(
  (agent as any).memory,
  (plan) => wsServer.broadcast({ type: "plan_update", taskId: "system", data: plan, timestamp: new Date() })
);

const healthDashboard = new HealthDashboardTool({
  terminal,
  tokenTracker,
  smartContext,
  logViewer: new (require("./tools/logViewerTool").LogViewerTool)(WORKSPACE_DIR),
});

// Start auto-checkpoints
checkpointTool.startAutoCheckpoints(5);

// ═══ AGENT MODE ROUTES ═══
app.get("/api/mode", (_, res) => {
  res.json({ mode: modeManager.getMode(), config: modeManager.getConfig(), features: modeManager.getFeatures() });
});

app.post("/api/mode", (req, res) => {
  const config = modeManager.setMode(req.body.mode);
  res.json({ mode: req.body.mode, config });
});

app.post("/api/mode/feature", (req, res) => {
  const { feature, value } = req.body;
  const result = modeManager.toggleFeature(feature, value);
  res.json({ feature, enabled: result });
});

app.post("/api/mode/estimate", (req, res) => {
  res.json(modeManager.estimateCost(req.body.complexity || "medium"));
});

// ═══ REQUEST QUEUE ROUTES ═══
app.post("/api/chat", (req, res) => {
  const { message, priority, context } = req.body;
  const request = requestQueue.enqueue(message, { priority, context });
  res.json(request);
});

app.get("/api/chat/queue", (_, res) => {
  res.json({ queue: requestQueue.getQueue(), current: requestQueue.getCurrent(), stats: requestQueue.getStats() });
});

app.delete("/api/chat/queue/:id", (req, res) => {
  res.json({ cancelled: requestQueue.cancel(req.params.id) });
});

// ═══ PLAN MODE ROUTES ═══
app.post("/api/plan", async (req, res) => {
  const plan = await planMode.createPlan(req.body.task);
  res.json(plan);
});

app.get("/api/plan", (_, res) => {
  res.json(planMode.getPlan());
});

app.post("/api/plan/approve", (req, res) => {
  if (req.body.taskId) {
    res.json({ approved: planMode.approveTask(req.body.taskId) });
  } else {
    planMode.approveAll();
    res.json({ approved: true, ready: planMode.isReadyToExecute() });
  }
});

app.post("/api/plan/reject", (req, res) => {
  res.json({ rejected: planMode.rejectTask(req.body.taskId, req.body.reason) });
});

app.post("/api/plan/modify", (req, res) => {
  res.json({ modified: planMode.modifyTask(req.body.taskId, req.body.changes) });
});

app.post("/api/plan/answer", async (req, res) => {
  const plan = await planMode.answerQuestions(req.body.answers);
  res.json(plan);
});

// ═══ CHECKPOINT ROUTES ═══
app.get("/api/timeline", (_, res) => {
  res.json({ timeline: checkpointTool.getTimeline() });
});

app.post("/api/timeline/travel", async (req, res) => {
  const result = await checkpointTool.travelTo(req.body.checkpointId);
  res.json(result);
});

app.get("/api/timeline/:id/diff", async (req, res) => {
  const diff = await checkpointTool.compareTo(req.params.id);
  res.json({ diff });
});

app.post("/api/checkpoint", async (req, res) => {
  const snap = await checkpointTool.createCheckpoint("manual", req.body.description);
  res.json(snap);
});

// ═══ HOTFIX ROUTES ═══
app.post("/api/flow/hotfix", async (req, res) => {
  const result = await hotfixFlow.hotfix(req.body.issue);
  res.json(result);
});

// ═══ RULES ROUTES ═══
app.get("/api/rules", (_, res) => {
  res.json(rulesSyncTool.getRules());
});

app.put("/api/rules", (req, res) => {
  rulesSyncTool.updateRules(req.body);
  res.json({ success: true, rules: rulesSyncTool.getRules() });
});

app.post("/api/rules/custom", (req, res) => {
  rulesSyncTool.setCustomRule(req.body.key, req.body.value);
  res.json({ success: true });
});

// ═══ TOKEN TRACKING ROUTES ═══
app.get("/api/tokens", (_, res) => {
  res.json(tokenTracker.getStats());
});

app.get("/api/tokens/recent", (req, res) => {
  res.json(tokenTracker.getRecent(parseInt(req.query.limit as string) || 50));
});

// ═══ HEALTH DASHBOARD ═══
app.use("/api/dashboard", healthDashboard.getRoutes());
