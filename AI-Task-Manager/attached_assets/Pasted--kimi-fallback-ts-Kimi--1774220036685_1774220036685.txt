export interface FallbackProvider {
  name:     string;
  baseUrl:  string;
  apiKey:   string;
  model:    string;
  enabled:  boolean;
}

export const PROVIDERS: FallbackProvider[] = [
  {
    name:    'kimi',
    baseUrl: 'https://api.moonshot.cn/v1',
    apiKey:  process.env.KIMI_API_KEY ?? '',
    model:   'moonshot-v1-8k',
    enabled: true,
  },
  {
    name:    'openai',
    baseUrl: 'https://api.openai.com/v1',
    apiKey:  process.env.OPENAI_API_KEY ?? '',
    model:   'gpt-4o-mini',
    enabled: !!process.env.OPENAI_API_KEY,
  },
  {
    name:    'groq',
    baseUrl: 'https://api.groq.com/openai/v1',
    apiKey:  process.env.GROQ_API_KEY ?? '',
    model:   'llama-3.3-70b-versatile',
    enabled: !!process.env.GROQ_API_KEY,
  },
];

export async function callWithFallback(
  messages: { role: string; content: string }[],
  options: { maxTokens?: number; temperature?: number; timeoutMs?: number } = {}
): Promise<{ result: string; provider: string }> {
  const active = PROVIDERS.filter(p => p.enabled && p.apiKey);

  for (const provider of active) {
    try {
      console.log(`[Fallback] Trying ${provider.name}...`);
      const res = await fetch(`${provider.baseUrl}/chat/completions`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${provider.apiKey}` },
        body: JSON.stringify({
          model:       provider.model,
          messages,
          max_tokens:  options.maxTokens  ?? 2048,
          temperature: options.temperature ?? 0.3,
        }),
        signal: AbortSignal.timeout(options.timeoutMs ?? 30_000),
      });

      if (!res.ok) {
        const text = await res.text();
        console.warn(`[Fallback] ${provider.name} failed: ${res.status} ${text.slice(0,100)}`);
        continue;
      }

      const data = await res.json() as any;
      const content = data?.choices?.[0]?.message?.content?.trim();
      if (!content) { console.warn(`[Fallback] ${provider.name} returned empty`); continue; }

      console.log(`[Fallback] ✅ ${provider.name} succeeded`);
      return { result: content, provider: provider.name };

    } catch (e) {
      console.warn(`[Fallback] ${provider.name} error:`, (e as Error).message);
    }
  }

  throw new Error(`All providers failed: ${active.map(p => p.name).join(', ')}`);
}
